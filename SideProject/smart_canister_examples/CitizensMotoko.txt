import Map "mo:base/HashMap";
import Text "mo:base/Text";
import Nat "mo:base/Nat";

actor {
  type TC = Text;

  type Citizen = {
    name : Text;
    surname : Text;
    birthday : Nat;
    marriage : Bool;
  };

  let all_citizens = Map.HashMap<TC, Citizen>(0, Text.equal, Text.hash);

  public func newBorn(tc : TC, citizen : Citizen) : async () {
    all_citizens.put(tc, citizen);
  };

  public query func lookup(tc : TC) : async ?Citizen {
    all_citizens.get(tc);
  };

};
